        // ---------- Scoring ----------
        function calculateScores() {
            const checks = document.querySelectorAll('.riasec-chk');
            const scores = { R:0, I:0, A:0, S:0, E:0, C:0 };
            checks.forEach(chk => {
                if (chk.checked) {
                    const type = chk.dataset.type;
                    scores[type]++;
                }
            });
            return scores;
        }

        // ---------- Get top 3 letters (order by score desc, tie break by original order) ----------
        function getTop3(scores) {
            const entries = Object.entries(scores);
            entries.sort((a, b) => {
                if (b[1] !== a[1]) return b[1] - a[1];
                // tie: prefer order R,I,A,S,E,C
                const order = {R:0, I:1, A:2, S:3, E:4, C:5};
                return order[a[0]] - order[b[0]];
            });
            return entries.slice(0,3).map(e => e[0]);
        }

        // ---------- Fuzzy match fields: compute relevance score ----------
        function scoreField(field, top3) {
            const [t1, t2, t3] = top3;
            const f1 = field.first;
            const f2 = field.second;
            const f3 = field.third || ''; // may be empty string

            let score = 0;

            // Exact position matches (high weight)
            if (f1 === t1) score += 5;
            if (f2 === t2) score += 4;
            if (f3 && f3 === t3) score += 3;

            // Other matches (lower weight) – count each matching letter in any position
            // but avoid double counting already matched positions (we can just add small bonus)
            const userSet = new Set([t1, t2, t3]);
            const fieldSet = new Set([f1, f2, f3].filter(l => l !== ''));
            // Intersection size gives number of common letters
            const common = [...userSet].filter(l => fieldSet.has(l)).length;
            // Base score already counted some, so we add remaining as bonus
            // Let's add 1 per common letter that wasn't already matched in exact position
            // But simpler: add 1 for each common letter (capped to avoid overcount)
            // This rewards fields that share multiple interests
            score += common * 1; // simple bonus

            return score;
        }

        // ---------- Find and sort fields by relevance ----------
        function findRelevantFields(top3) {
            const scored = fieldData.map(field => ({
                ...field,
                score: scoreField(field, top3)
            }));
            // sort descending by score
            scored.sort((a, b) => b.score - a.score);
            // keep only those with score > 0 (or you can show all, but zero means no match at all)
            return scored.filter(f => f.score > 0);
        }

            { left: 'Clear', right: 'Confusing' },
            { left: 'Fast', right: 'Slow' },
            { left: 'Helpful', right: 'Not helpful' },
            { left: 'Accurate', right: 'Inaccurate' },
            { left: 'Relevant', right: 'Irrelevant' }
        ];

        function renderFeedback() {
            const container = document.getElementById('feedbackItems');
            container.innerHTML = '';
            feedbackPairs.forEach((pair, idx) => {
                const itemDiv = document.createElement('div');
                itemDiv.className = 'feedback-item';
                let radios = '';
                for (let i = 1; i <= 7; i++) {
                    radios += `<label><input type="radio" name="fb${idx+1}" value="${i}" required> ${i}</label>`;
                }
                itemDiv.innerHTML = `
                    <div style="flex:1; min-width:200px;">
                        <strong>${pair.left}</strong>  &nbsp;–&nbsp;  <strong>${pair.right}</strong>
                    </div>
                    <div class="radio-group">
                        ${radios}
                    </div>
                `;
                container.appendChild(itemDiv);
            });
        }
        renderFeedback();

        // ---------- Submit to Apps Script ----------
        document.getElementById('submitBtn').addEventListener('click', () => {
            const name = document.getElementById('studentName').value.trim();
            if (!name) { alert('Name is missing.'); return; }

            // Check all feedback radios are selected
            let feedbackValues = [];
            for (let i = 1; i <= 6; i++) {
                const radios = document.getElementsByName(`fb${i}`);
                let selected = null;
                for (let r of radios) {
                    if (r.checked) { selected = r.value; break; }
                }
                if (!selected) {
                    alert(`Please answer feedback item ${i}.`);
                    return;
                }
                feedbackValues.push(parseInt(selected));
            }

            if (!lastScores || !lastTop3) {
                alert('Please calculate your RIASEC code first.');
                return;
            }

            const payload = {
                timestamp: new Date().toISOString(),
                name: name,
                scores: lastScores,
                top3: lastTop3,
                feedback: feedbackValues
            };

            // ---- REPLACE WITH YOUR APPS SCRIPT URL ----
            const scriptURL = 'https://script.google.com/macros/s/AKfycbyI59cHWtybgeP4eqvCC-pTo5MuHreuCjlmj3gXQdxBI2ZDEzrNkCHUxjiOAlApZ8XRWw/exec';

            document.getElementById('submitStatus').textContent = 'Submitting...';
            fetch(scriptURL, {
                method: 'POST',
                mode: 'no-cors',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            })
            .then(() => {
                document.getElementById('submitStatus').textContent = '✓ Data saved. Preparing print...';
                setTimeout(() => {
                    window.print();
                }, 500);
            })
            .catch(err => {
                document.getElementById('submitStatus').textContent = 'Error: ' + err.message;
            });
        });
    </script>
