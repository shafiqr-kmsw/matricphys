    <script>
        let cachedDataset = [];
        let activeChartObject = null;

        function searchStudent() {
            const idValue = document.getElementById('studentId').value.trim();
            if (!idValue) {
                displayError('Validation Error: Input space cannot evaluate as empty.');
                return;
            }

            hideError();
            document.getElementById('dashboardFrame').classList.add('hidden');
            document.getElementById('filterContainer').classList.add('invisible');
            document.getElementById('loader').classList.remove('hidden');
            document.getElementById('searchBtn').disabled = true;

            google.script.run
                .withSuccessHandler(function(response) {
                    document.getElementById('loader').classList.add('hidden');
                    document.getElementById('searchBtn').disabled = false;

                    if (!response || response.length === 0) {
                        displayError('No matching data logs register against ID: ' + idValue);
                        return;
                    }

                    cachedDataset = response;
                    initializeDashboard(response);
                })
                .withFailureHandler(function(err) {
                    document.getElementById('loader').classList.add('hidden');
                    document.getElementById('searchBtn').disabled = false;
                    displayError('System Processing Interruption: ' + err.message);
                })
                .getStudentData(idValue);
        }

        function initializeDashboard(records) {
            const foundationalRow = records[0];
            document.getElementById('metaId').innerText = foundationalRow.id;
            document.getElementById('metaName').innerText = foundationalRow.name;
            document.getElementById('metaClass').innerText = foundationalRow.class;

            const selector = document.getElementById('subjectSelect');
            selector.innerHTML = '<option value="overall">Overall Results</option>';
            
            records.forEach(row => {
                const item = document.createElement('option');
                item.value = row.subject;
                item.innerText = row.subject;
                selector.appendChild(item);
            });

            document.getElementById('filterContainer').classList.remove('invisible');
            document.getElementById('dashboardFrame').classList.remove('hidden');
            
            updateView();
        }

        function updateView() {
            const focusFilter = document.getElementById('subjectSelect').value;
            const targetTableBody = document.getElementById('matrixTableRows');
            targetTableBody.innerHTML = '';

            let filteredSubset = (focusFilter === 'overall') 
                ? cachedDataset 
                : cachedDataset.filter(record => record.subject === focusFilter);

            filteredSubset.forEach(row => {
                const mark1 = Number(row.a1) || 0;
                const mark2 = Number(row.a2) || 0;
                const mark3 = Number(row.a3) || 0;
                const mathematicalMean = ((mark1 + mark2 + mark3) / 3).toFixed(1);

                const tableRowElement = document.createElement('tr');
                tableRowElement.className = "hover:bg-gray-50/50 transition";
                tableRowElement.innerHTML = `
                    <td class="px-4 py-3.5 font-semibold text-gray-900">${row.subject}</td>
                    <td class="px-4 py-3.5 text-center text-gray-600">${mark1}</td>
                    <td class="px-4 py-3.5 text-center text-gray-600">${mark2}</td>
                    <td class="px-4 py-3.5 text-center text-gray-600">${mark3}</td>
                    <td class="px-4 py-3.5 text-center font-bold text-blue-600 bg-blue-50/50 rounded-lg">${mathematicalMean}%</td>
                `;
                targetTableBody.appendChild(tableRowElement);
            });

            generateVisualization(focusFilter, filteredSubset);
        }

        function generateVisualization(viewMode, metricsArray) {
            setTimeout(function() {
                try {
                    // Double check if the library actually loaded this time
                    if (typeof Chart === 'undefined') {
                        throw new Error("The Chart framework could not download. Your internet firewall is blocking the chart script file.");
                    }

                    const wrapper = document.getElementById('chartWrapper');
                    wrapper.innerHTML = '<canvas id="canvasChartInstance"></canvas>';
                    const renderContext = document.getElementById('canvasChartInstance').getContext('2d');

                    if (activeChartObject) {
                        activeChartObject.destroy();
                    }

                    const chartColorPalettes = [
                        { stroke: 'rgba(37, 99, 235, 1)', fill: 'rgba(37, 99, 235, 0.1)' },
                        { stroke: 'rgba(219, 39, 119, 1)', fill: 'rgba(219, 39, 119, 0.1)' },
                        { stroke: 'rgba(5, 150, 105, 1)', fill: 'rgba(5, 150, 105, 0.1)' },
                        { stroke: 'rgba(217, 119, 6, 1)', fill: 'rgba(217, 119, 6, 0.1)' }
                    ];

                    let activeConfig = {};

                    if (viewMode === 'overall') {
                        const trackingDatasets = metricsArray.map((row, index) => {
                            const tint = chartColorPalettes[index % chartColorPalettes.length];
                            return {
                                label: row.subject,
                                data: [Number(row.a1) || 0, Number(row.a2) || 0, Number(row.a3) || 0],
                                borderColor: tint.stroke,
                                backgroundColor: tint.fill,
                                tension: 0.2,
                                borderWidth: 3,
                                pointRadius: 5,
                                pointHoverRadius: 7
                            };
                        });

                        activeConfig = {
                            type: 'line',
                            data: {
                                labels: ['Assessment 1', 'Assessment 2', 'Assessment 3'],
                                datasets: trackingDatasets
                            },
                            options: {
                                responsive: true,
                                maintainAspectRatio: false,
                                scales: {
                                    y: {
                                        min: 0,
                                        max: 100,
                                        title: { display: true, text: 'Scores (%)', font: { weight: 'bold' } }
                                    }
                                }
                            }
                        };
                    } else {
                        const targetedSubjectRecord = metricsArray[0];
                        activeConfig = {
                            type: 'bar',
                            data: {
                                labels: ['Assessment 1', 'Assessment 2', 'Assessment 3'],
                                datasets: [{
                                    label: targetedSubjectRecord.subject + ' Score Profile',
                                    data: [
                                        Number(targetedSubjectRecord.a1) || 0, 
                                        Number(targetedSubjectRecord.a2) || 0, 
                                        Number(targetedSubjectRecord.a3) || 0
                                    ],
                                    backgroundColor: ['rgba(37, 99, 235, 0.75)', 'rgba(5, 150, 105, 0.75)', 'rgba(219, 39, 119, 0.75)'],
                                    borderColor: ['rgba(37, 99, 235, 1)', 'rgba(5, 150, 105, 1)', 'rgba(219, 39, 119, 1)'],
                                    borderWidth: 1.5,
                                    borderRadius: 6
                                }]
                            },
                            options: {
                                responsive: true,
                                maintainAspectRatio: false,
                                scales: {
                                    y: {
                                        min: 0,
                                        max: 100,
                                        title: { display: true, text: 'Scores (%)', font: { weight: 'bold' } }
                                    }
                                }
                            }
                        };
                    }

                    activeChartObject = new Chart(renderContext, activeConfig);
                } catch (err) {
                    displayError('Visualization Rendering Crash: ' + err.message);
                }
            }, 100);
        }

        function displayError(msg) {
            const element = document.getElementById('errorAlert');
            document.getElementById('errorText').innerText = msg;
            element.classList.remove('hidden');
        }

        function hideError() {
            document.getElementById('errorAlert').classList.add('hidden');
        }
    </script>
